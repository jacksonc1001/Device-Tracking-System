	@I_LotNumber VARCHAR(50),
	@I_Operation VARCHAR(50),
	@I_Equipment VARCHAR(50),
	@O_Success INT OUTPUT,
	@O_ErrTxt VARCHAR(250) OUTPUT

AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	DECLARE @ERR_NUM INT, @ERR_MSG NVARCHAR(4000)
	DECLARE @DeviceCount INT

	SET @DeviceCount = 0
	SET @O_Success = 0
	SET @O_ErrTxt = ''

	BEGIN TRY
		SELECT @DeviceCount = COUNT(di.DeviceId)
		FROM DIS_DeviceInformation di
		INNER JOIN DIS_Lot l ON l.LotId = di.LotId
		INNER JOIN DIS_Equipment e ON e.EqpId = di.EqpId
		WHERE l.LotNumber = @I_LotNumber
		AND di.OpId <= @I_Operation AND di.Active = 1

		IF (@DeviceCount = 0)
			BEGIN
				SET @O_Success = 1
				SET @O_ErrTxt = 'No Device Found'
			END
		ELSE
			BEGIN
				SELECT di.DeviceId, di.AliasId, di.TargetSubstrateId, di.TargetPositionId,
				di.ToX, di.ToY, di.Result, di.BinCode, di.Pick, di.BinDesc
				FROM DIS_DeviceInformation di
				INNER JOIN DIS_Lot l ON l.LotId = di.LotId
				INNER JOIN DIS_Equipment e ON e.EqpId = di.EqpId
				WHERE l.LotNumber = @I_LotNumber
				AND di.OpId <= @I_Operation AND di.Active = 1
			END
	END TRY
	BEGIN CATCH
		SET @O_Success = -1
		SELECT @ERR_NUM = ERROR_NUMBER(), @ERR_MSG = ERROR_MESSAGE()
		SET @O_ErrTxt = @ERR_MSG
	END CATCH
END
GO

CREATE PROCEDURE [dbo].[PR_DIS_UpdateDeviceInfo] 
	@Tbl_DeviceInformationList DIS_DeviceInformationList READONLY,
	@I_LotNumber VARCHAR(50),
	@I_Operation INT,
	@I_Equipment VARCHAR(250),
	@O_Success INT OUTPUT,
	@O_ErrTxt VARCHAR(250) OUTPUT

AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	DECLARE @ERR_NUM INT, @ERR_MSG NVARCHAR(4000)
	DECLARE @TRANS_NAME VARCHAR(50)
	DECLARE @CurrDateTime DATETIME

	SET NOCOUNT ON;
	SET @TRANS_NAME = 'UpdateDeviceInfo'
	SET @O_Success = 0
	SET @CurrDateTime = GETDATE()

	BEGIN TRY
		BEGIN TRANSACTION @TRANS_NAME
			
			SELECT * INTO #TMPDeviceInfo FROM @Tbl_DeviceInformationList
			
			UPDATE DIS_DeviceInformation
			SET 
			Active = 0
			FROM DIS_DeviceInformation di
			INNER JOIN #TMPDeviceInfo tmp ON tmp.DeviceId = di.DeviceId
			INNER JOIN DIS_Lot l ON l.LotId = di.LotId
			WHERE l.LotNumber = @I_LotNumber AND di.OpId <= @I_Operation

			INSERT INTO DIS_DeviceInformation
			(DeviceId,AliasId,SourceSubstrateId,TargetSubstrateId,SourcePositionId,TargetPositionId,FromX,FromY,ToX
			,ToY,Result,BinCode,BinDesc,SAPBarcode,Pick,Active,LotId,OpId,InsertDateTime,EqpId)
			SELECT
			tpi.DeviceId, tpi.AliasId, tpi.SourceSubstrateId, tpi.TargetSubstrateId, 
			tpi.SourcePositionId, tpi.TargetPositionId, tpi.FromX, tpi.FromY, tpi.ToX
			, tpi.ToY, 
			Result = CASE WHEN tpi.Result = 'UNKNOWN' THEN 'PREV FAIL' ELSE tpi.Result END,
			tpi.BinCode, tpi.BinDesc, '', Pick = CASE WHEN tpi.Result = 'UNKNOWN' OR tpi.Result = 'FAIL' THEN 0 ELSE 1 END,
			1, (SELECT LotId FROM DIS_Lot WHERE LotNumber = @I_LotNumber), @I_Operation, @CurrDateTime, 
			(SELECT EqpId FROM DIS_Equipment WHERE EquipmentName = @I_Equipment)
			FROM  #TMPDeviceInfo tpi

			DROP TABLE #TMPDeviceInfo

		COMMIT TRANSACTION @TRANS_NAME

	END TRY
	BEGIN CATCH
		SET @O_Success = -1
		SELECT @ERR_NUM = ERROR_NUMBER(), @ERR_MSG = ERROR_MESSAGE()
		ROLLBACK TRANSACTION @TRANS_NAME
		SET @O_ErrTxt = @ERR_MSG
	END CATCH
END

-- Create date: 02/03/2022
-- Description:	To Merge the Device Information 
-- =============================================
CREATE PROCEDURE [dbo].[PR_DIS_UpdateMergeDevice]
	@I_MotherLot VARCHAR(50),
	@I_ChildLot VARCHAR(50),
	@I_ChildLot2 VARCHAR(50),
	@I_ChildLot3 VARCHAR(50),
	@I_ChildLot4 VARCHAR(50),
	@I_Equipment VARCHAR(50),
	@I_Operation INT,
	@O_Success INT OUTPUT,
	@O_ErrTxt VARCHAR(250) OUTPUT

AS
BEGIN
	DECLARE @ERR_NUM INT, @ERR_MSG NVARCHAR(4000)
	DECLARE @TRANS_NAME VARCHAR(50)
	DECLARE @CurrDate DATETIME

	SET NOCOUNT ON;
	SET @O_Success = 0
	SET @O_ErrTxt = ''
	SET @CurrDate = GETDATE()
	SET @TRANS_NAME = 'MergeDevice'

	BEGIN TRY
		BEGIN TRANSACTION @TRANS_NAME
			
			SELECT * INTO #TMPMergeDevicesList FROM DIS_DeviceInformation di
			WHERE di.LotId in (SELECT LotId FROM DIS_Lot WHERE LotNumber = @I_MotherLot OR LotNumber = @I_ChildLot
			OR LotNumber = @I_ChildLot2 OR LotNumber = @I_ChildLot3
			OR LotNumber = @I_ChildLot4) 
			AND di.Active = 1

			UPDATE DIS_DeviceInformation
			SET 
			Active = 0
			FROM DIS_DeviceInformation di
			INNER JOIN DIS_Lot l ON l.LotId = di.LotId
			WHERE (l.LotNumber = @I_MotherLot OR l.LotNumber = @I_ChildLot
			OR l.LotNumber = @I_ChildLot2 OR l.LotNumber = @I_ChildLot3
			OR l.LotNumber = @I_ChildLot4)
			AND di.Active = 1

			INSERT INTO DIS_DeviceInformation
			(DeviceId,AliasId,SourceSubstrateId,TargetSubstrateId,SourcePositionId,TargetPositionId,FromX,FromY,ToX
			,ToY,Result,BinCode,BinDesc,SAPBarcode,Pick,Active,LotId,OpId,InsertDateTime,EqpId)
			SELECT
			tpl.DeviceId, tpl.AliasId, tpl.SourceSubstrateId, tpl.TargetSubstrateId, 
			tpl.SourcePositionId, tpl.TargetPositionId, tpl.FromX, tpl.FromY, tpl.ToX
			, tpl.ToY, tpl.Result, tpl.BinCode, tpl.BinDesc, tpl.SAPBarcode, tpl.Pick,
			1, (SELECT LotId FROM DIS_Lot WHERE LotNumber = @I_MotherLot), tpl.OpId, @CurrDate, tpl.EqpId
			FROM  #TMPMergeDevicesList tpl
			ORDER BY tpl.Id ASC

			DROP TABLE #TMPMergeDevicesList

		COMMIT TRANSACTION @TRANS_NAME

	END TRY
	BEGIN CATCH
		SET @O_Success = -1
		SELECT @ERR_NUM = ERROR_NUMBER(), @ERR_MSG = ERROR_MESSAGE()
		ROLLBACK TRANSACTION @TRANS_NAME
		SET @O_ErrTxt = @ERR_MSG
	END CATCH
END
